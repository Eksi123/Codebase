#加载程序包
library(mFilter)

#虚拟经济指标数据预处理
#自定义数据
X1<-fictious_ts[,1]
#季度调整
lX1<-log(X1)
fit<-stl(lX1,s.window="period")
fit1<-fit$time.series[,2]+fit$time.series[,3]
adj_X1<-exp(fit1)
y_X1<-diff(log(adj_X1))
#HP滤波分解+提取周期波动项
tc_X1<-hpfilter(adj_X1,freq=1600,type="lambda",drift=FALSE)
c_X1<-tc_X1$cycle
